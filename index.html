Plateau d'Othello

packages = ["pygame-ce"] import pygame import asyncio async def main(): pygame.init() screen = pygame.display.set_mode((400, 400)) # 0=vide, 1=noir, 2=blanc board = [[0]*8 for _ in range(8)] board[3][3], board[4][4] = 2, 2 # Pions blancs board[3][4], board[4][3] = 1, 1 # Pions noirs while True: screen.fill((34, 139, 34)) # Vert plateau # Dessiner le quadrillage et les pions for r in range(8): for c in range(8): pygame.draw.rect(screen, (0,0,0), (c*50, r*50, 50, 50), 1) if board[r][c] == 1: pygame.draw.circle(screen, (0,0,0), (c*50+25, r*50+25), 20) elif board[r][c] == 2: pygame.draw.circle(screen, (255,255,255), (c*50+25, r*50+25), 20) pygame.display.flip() await asyncio.sleep(0.05) asyncio.ensure_future(main())
